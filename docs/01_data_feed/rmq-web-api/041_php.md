#### Usage Sample

```
<?php

require_once 'Requestor.php';

require_once 'RmqDataHandlerSample.php';

require_once 'IRmqDataHandler.php';

require_once 'RmqSubscribeClient.php';

error_reporting(E_ALL && ~E_WARNING);

date_default_timezone_set ( 'Asia/Yerevan' );

set_time_limit ( 0 );

//include default values

require('config.php');

$WebApiClient = new \WebApiClient($apiParams);

$Requestor = new \Requestor($WebApiClient);

$sports = $Requestor->getSports();

$competitions = $Requestor->getCompetitions();

$competition = $Requestor->getCompetitionById(29971);

$booking = $Requestor->getBookings(true);

$match = $Requestor->getMatchById(19824998,  true);

$dataSnapshot = $Requestor->getDataSnapshot(true, 1);

$sportOrder = $Requestor->getSportOrder();

//...

//class must implement IRmqDataHandler interface

$handler = new \RmqDataHandlerSample();

//RMQ SUBSCRIBE FOR UPDATES

$rmqClient = new RmqSubscribeClient($rmqParams, $handler);

$rmqClient->subscribe();
```
